import json
import requests
from langchain.chat_models import AzureChatOpenAI
from langchain.prompts import PromptTemplate
from langchain.schema.output_parser import StrOutputParser
from langchain.schema.runnable import RunnablePassthrough, RunnableLambda
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.chains import LLMChain
from langchain.agents import Tool, initialize_agent
from langchain.memory import ConversationBufferMemory

# === MCP Server Setup ===
MCP_URL = "https://your.mcp.server.url"  # âœ… replace with actual server
MCP_PAYLOAD = {
    "method": "tools/call",
    "params": {
        "name": "get-vendors",
        "arguments": {}
    }
}

# === Globals ===
mcp_raw_data = ""

# === Azure OpenAI Config ===
def get_openai_client():
    return AzureChatOpenAI(
        azure_endpoint="https://testopenaiassets.openai.azure.com",
        deployment_name="Fourth_Chatbot",
        openai_api_key="",  # ðŸ”‘ Add your Azure OpenAI key here
        openai_api_version="2024-08-01-preview",
        openai_api_type="azure",
        temperature=0.5,
        max_tokens=1500,
        model_kwargs={"top_p": 0.9}
    )

# === Tool 1: Step 1 â€” MCP Server Connection Check ===
def check_mcp_server(_: str) -> str:
    try:
        response = requests.get(MCP_URL, timeout=5)
        if response.status_code == 200:
            return "[SUCCESS] MCP server is reachable."
        return f"[FAILURE] MCP server responded with status code: {response.status_code}"
    except Exception as e:
        return f"[ERROR] Failed to connect to MCP server: {str(e)}"

# === Tool 2: Step 2 â€” Fetch Data from MCP ===
def fetch_mcp_data(_: str) -> str:
    global mcp_raw_data
    try:
        response = requests.post(MCP_URL, json=MCP_PAYLOAD, timeout=10)
        response.raise_for_status()
        mcp_raw_data = json.dumps(response.json(), indent=2)
        return "[SUCCESS] Data fetched from MCP server and stored."
    except Exception as e:
        return f"[ERROR] Unable to fetch data from MCP server: {str(e)}"

# === Tool 3: Step 3 â€” Format and Summarize ===
def summarize_mcp_data(_: str) -> str:
    global mcp_raw_data
    if not mcp_raw_data:
        return "[SKIP] No MCP data found. Please run fetch step first."

    try:
        splitter = RecursiveCharacterTextSplitter(chunk_size=1000, chunk_overlap=100)
        chunks = splitter.split_text(mcp_raw_data)

        client = get_openai_client()
        prompt = PromptTemplate(
            input_variables=["chunk"],
            template="""
You are an assistant. Summarize the following vendor-related JSON chunk in plain, user-friendly language.
Highlight:
- Key vendor info
- Any configurations or fields present
- Important values or relationships

{chunk}
"""
        )
        summarize_chain = LLMChain(llm=client, prompt=prompt)

        all_summaries = [summarize_chain.run({"chunk": chunk}) for chunk in chunks]
        return "\n\n".join(all_summaries)

    except Exception as e:
        return f"[ERROR] Failed to summarize MCP data: {str(e)}"

# === LangChain Agent Setup ===
def run_agent():
    memory = ConversationBufferMemory(memory_key="chat_history", return_messages=True)

    tools = [
        Tool(
            name="CheckMCPServer",
            func=check_mcp_server,
            description="Check if the MCP server URL is reachable (Step 1)."
        ),
        Tool(
            name="FetchMCPData",
            func=fetch_mcp_data,
            description="Fetch vendor data from MCP server using predefined payload (Step 2)."
        ),
        Tool(
            name="SummarizeMCPData",
            func=summarize_mcp_data,
            description="Summarize the vendor data retrieved from MCP for easy understanding (Step 3)."
        )
    ]

    agent = initialize_agent(
        tools=tools,
        llm=get_openai_client(),
        agent_type="openai-functions",
        memory=memory,
        verbose=True
    )
    return agent

# === Main Execution ===
if __name__ == "__main__":
    agent = run_agent()
    print(">> Starting MCP Data Processing...\n")
    agent.invoke("Check the MCP server, fetch vendor data, and summarize it for end users.")
