import os
import shutil
from pathlib import Path
import gradio as gr
import json
from langchain.prompts import PromptTemplate
from langchain_openai import AzureChatOpenAI

# === Azure OpenAI Client ===
def get_openai_client():
    return AzureChatOpenAI(
        azure_endpoint="https://testopenaiassets.openai.azure.com",
        deployment_name="gpt-4_complex_conversions",
        openai_api_key="",
        openai_api_version="2024-08-01-preview",
        openai_api_type="azure",
        temperature=0.3,
        max_tokens=4000,
    )

llm = get_openai_client()

# === Step 1: Extract project name + reference path ===
def extract_user_intent(user_input: str) -> dict:
    prompt = PromptTemplate(
        input_variables=["user_input"],
        template="""
From the following user request, extract two things:

1. new_project_name → the new project name to create  
2. reference_path → the reference project path provided  

Return ONLY valid JSON:
{{
  "new_project_name": "...",
  "reference_path": "..."
}}

User request:
{user_input}
"""
    )
    response = (prompt | llm).invoke({"user_input": user_input})
    return json.loads(response.content.strip())

# === Step 2: AI Rewriter for files ===
def ai_rewrite_file(source: str, new_name: str) -> str:
    prompt = PromptTemplate(
        input_variables=["source", "new_name"],
        template="""
You are refactoring a full Java project.

Rules:
- Detect the old project name (from packages, class names, pom.xml, configs, etc.).
- Replace it consistently with {new_name}.
- Keep structure, comments, imports, annotations, and formatting intact.
- Do not drop any content.
- Only output the updated file text, nothing else.

Source file:
{source}
"""
    )
    response = (prompt | llm).invoke({"source": source, "new_name": new_name})
    return response.content.strip()

# === Step 3: Rewrite full project ===
def rewrite_project(reference_path: str, new_project_name: str) -> str:
    ref_path = Path(reference_path)
    if not ref_path.exists():
        return f"[ERROR] Reference project not found: {reference_path}"

    target_base = Path("C:/Users/rdamera/generated_projects")
    target_path = target_base / new_project_name
    if target_path.exists():
        shutil.rmtree(target_path)
    shutil.copytree(ref_path, target_path)

    # Walk through files and rewrite text ones
    for file_path in target_path.rglob("*"):
        if file_path.is_file() and file_path.suffix.lower() in [".java", ".xml", ".yml", ".yaml", ".properties", ".json", ".md", ".txt"]:
            try:
                source = file_path.read_text(encoding="utf-8", errors="ignore")
                updated = ai_rewrite_file(source, new_project_name)
                file_path.write_text(updated, encoding="utf-8")
            except Exception as e:
                print(f"[WARN] Skipping {file_path}: {e}")

    return str(target_path)

# === Main Orchestration ===
def process_java_project(user_input: str, history):
    try:
        intent = extract_user_intent(user_input)
        reference_path = intent["reference_path"]
        new_project_name = intent["new_project_name"]

        new_path = rewrite_project(reference_path, new_project_name)
        return f"[SUCCESS] Project '{new_project_name}' created at {new_path}"

    except Exception as e:
        return f"[ERROR] {str(e)}"

# === Gradio UI ===
chatbot_ui = gr.ChatInterface(
    fn=process_java_project,
    title="AI Java Project Generator",
    description="Provide a reference project path and a new project name. AI will regenerate the full project with the same structure but renamed.",
    theme="default",
)

if __name__ == "__main__":
    chatbot_ui.launch()
