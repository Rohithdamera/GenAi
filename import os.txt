from mcp import ClientSession
from mcp.client.sse import sse_client
 
async def main() -> None:
    # URL for the SSE stream
    url = "https://employee-mcp-v1-6b0n96.dw4w1g-2.gbr-e1.cloudhub.io/sse"
 
    # Prompt user for the tool name
    tool_name = input("Enter the tool name to call (e.g., get-vendors): ")
 
    async with sse_client(url=url) as streams:
        async with ClientSession(streams[0], streams[1]) as session:
            await session.initialize()
 
            tools = await session.list_tools()
            print("Available tools:", tools)
            # Call the tool using the user-defined tool name
            result = await session.call_tool(tool_name, {
                                            "location": "Bangalore"
            })

        print("Tool result:", result)
await main()
